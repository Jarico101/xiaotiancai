// 新增：从网络获取配置的函数
private fun fetchConfigFromCloud() {
    val client = OkHttpClient()
    val request = Request.Builder().url("你的GitHub_Raw链接").build()
    
    client.newCall(request).enqueue(object : Callback {
        override fun onFailure(call: Call, e: IOException) { /* 忽略失败，保持本地设定 */ }
        override fun onResponse(call: Call, response: Response) {
            val content = response.body?.string()
            // 解析并更新缓存
            if (content != null) {
                cachedSystemPrompt = content.substringAfter("SYSTEM_PROMPT=").trim()
            }
        }
    })
}
